    <script>
        const map = {
            'a': '🍎', 'b': '🥓', 'c': '🧀', 'd': '🍩', 'e': '🥚', 'f': '🍟', 'g': '🍇', 'h': '🍯', 'i': '🍦', 'j': '🍭',
            'k': '🥝', 'l': '🍋', 'm': '🍄', 'n': '🥜', 'o': '🍊', 'p': '🍕', 'q': '🍳', 'r': '🍙', 's': '🍓', 't': '🌮',
            'u': '🍲', 'v': '🥦', 'w': '🥨', 'x': '🍬', 'y': '🍿', 'z': '🍱', ' ': '🔳'
        };
        const rev = Object.fromEntries(Object.entries(map).map(([k, v]) => [v, k]));
        
        // This variable will store the password you set while encrypting
        let currentSalt = "";

        function run(mode) {
            const input = document.getElementById('inputField').value;
            const passInput = document.getElementById('passField').value.trim().toLowerCase();
            const status = document.getElementById('status');
            const output = document.getElementById('outputField');
            
            if (!input) return;

            if (mode === 'encrypt') {
                if (!passInput) {
                    alert("Set a password first so you can unlock it later!");
                    return;
                }
                currentSalt = passInput; // Save whatever was typed in the box
                let res = '';
                for (let c of input.toLowerCase()) res += map[c] || c;
                output.value = res;
                status.innerText = "LOCKED WITH: " + currentSalt.toUpperCase();
            } else {
                // If the user hasn't set a password in the session, we use "salt" as default
                const activePass = currentSalt || "salt"; 
                
                if (passInput !== activePass) {
                    status.innerText = "WRONG INGREDIENT! 🌶️";
                    output.value = "";
                    return;
                }
                let res = '';
                const chars = [...input];
                for (let c of chars) res += rev[c] || c;
                output.value = res;
                status.innerText = "UNLOCKED 🔓";
            }
        }

        function copyText() {
            const el = document.getElementById('outputField');
            el.select();
            navigator.clipboard.writeText(el.value);
            document.getElementById('status').innerText = "COPIED! 📋";
        }
    </script>
